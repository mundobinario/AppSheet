RentaSalida
EquipoID
SELECT
(
   Equipos[EquipoID]
   ,NOT
   (
     IN
     (
        [EquipoID]
        ,SELECT
        (
           RentaSalida[EquipoID]
           ,[_THISROW].[RentaID]=[RentaID]
        )
     )     
   )
)

TecnicoID
SELECT
(
   Tecnico[TecnicoID]
   ,IN([TecnicoPuesto], { "Supervisor", "Tecnico" })
)



RentaEntrada
EquipoID
SELECT
(
    RentaSalida[EquipoID]
    ,AND
    (
      [RentaID]= [_THISROW].[RentaID]
      ,NOT
      (
        IN
        (
           [EquipoID]
           ,SELECT
           (
               RentaEntrada[EquipoID]
               ,[RentaID]= [_THISROW].[RentaID]
           ) 
        )
      )
    )
)

Tecnico
Show if
IN(LOOKUP(USEREMAIL(), "Tecnico", "TecnicoCorreo", "TecnicoPuesto"), {"Supervisor" })
